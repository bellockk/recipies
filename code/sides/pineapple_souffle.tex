
\begin{recipe}{Pineapple Souffle}{serves 8}{1 hours}

  \ingredient[1]{cup}{butter}
  \ingredient[2]{cups}{sugar}
  Cream together.
  \ingredient[8]{}{eggs}
  \ingredient[2]{20oz. Cans}{crushed pineapple}
  Drain pineapple, and add with eggs to mixture.
  \ingredient[10]{slices}{fresh bread}
  Fold into mixture cubed bread.  Pour into a 3\nicefrac{1}{2} quart 
  casserole dish and bake 45 minutes at 350\0F.  Top will be golden and 
  crusty when done.
\end{recipe}
