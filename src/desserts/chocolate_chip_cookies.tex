\begin{recipe}{Chocolate Chip Cookies}{Makes 18 big cookies}{1 hour}
  Preheat oven to 325 degrees F (165 degrees C).  Line cookie sheets with parchment paper.
  \ingredient[2]{cups}{all-purpose flour}
  \ingredient[\nicefrac{1}{2}]{teaspoon}{baking soda}
  \ingredient[\nicefrac{1}{2}]{teaspoon}{salt}
  Sift together, set aside.
  \ingredient[\nicefrac{3}{4}]{cup}{unsalted butter, melted}
  \ingredient[1]{cup}{packed brown sugar}
  \ingredient[\nicefrac{1}{2}]{cup}{white sugar}
  Cream together.
  \ingredient[1]{tablespoon}{vanilla extract}
  \ingredient[1]{}{egg}
  \ingredient[1]{}{egg yolk}
  Beat in until light and creamy.  Mix the sifted ingredients until just blended.
  \ingredient[2]{cups}{semisweet chocolate chips}
  Stir in by hand.
  
  Drop cookie dough \nicefrac{1}{4} cup at a time onto the prepared cookie sheets.  Cookies should be at least 3 inches apart.  Bake for 15--17 minutes or until the egdes are just lightly toasted.  Cool on baking sheets for a few minutes before transferring to wire rracks to cool completely.
\end{recipe}
