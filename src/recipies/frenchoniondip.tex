
\begin{recipe}{French Onion Dip}{serves 8}{1 hours}
  \ingredient[1]{cup}{sour creme}
  \ingredient[3]{tbsp}{minced onions}
  \ingredient[1]{tbsp}{mayonaise}
  \ingredient[1/2]{tsp}{celery salt}
  \ingredient[1/2]{tsp}{dried dill weed}
  \ingredient[1/2]{tsp}{onion salt}
  \ingredient[1/4]{tsp}{garlic salt}
  \ingredient[1/4]{tsp}{mustard powder}
  \ingredient[1/4]{tsp}{black pepper}
Combine ingredients and serve.  Best if prepared the day before and left to sit in the refridgerator overnight.
\end{recipe}
