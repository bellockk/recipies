\begin{recipe}{Chicken Tikka Masala}{serves 4}{2 hours}
  \ingredient[1]{pound}{boneless skinless chicken thigh portions}
  Cut into small bite sized pieces.
  \ingredient[7]{ounces}{plain greek yogurt}
  \ingredient[1]{tablespoon}{garam masala}
  \ingredient[1]{tablespoon}{fresh lemon juice}
  \ingredient[1]{teaspoon}{ground black pepper}
  \ingredient[\nicefrac{1}{4}]{teaspoon}{ground ginger}
  Mix ingredients and cover chicken pieces so they are well coated.  Let sit for at least one hour, but overnight is preferred.
  \ingredient[15]{ounces}{tomato sauce}
  \ingredient[1\nicefrac{1}{2}]{cups}{heavy whipping cream}
  \ingredient[5]{cloves}{garlic}
  \ingredient[4]{teaspoons}{garam masala}
  \ingredient[\nicefrac{1}{2}]{teaspoon}{paprika}
  \ingredient[\nicefrac{1}{2}]{teaspoon}{table salt}
  \ingredient[\nicefrac{1}{2}]{teaspoon}{ground cayenne}
  Mix ingredients into Vitamix Blender and start soup cycle.
  When done, add chicken and serve with fresh naan.
\end{recipe}
