\begin{recipe}{Pineapple Souffle}{serves 8}{1 hours}

  \ingredient[1]{cup}{brown sugar}
  \ingredient[2]{cups}{Dale's Original Seasoning}
  \ingredient[3]{teaspoons}{ground black pepper}
  \ingredient[6]{teaspoons}{ground paprika}
  Blend together.
  \ingredient[5]{lbs}{Top Round Steak}
  Mix in a gallon zip lock bag and let marinade overnight.

\end{recipe}
